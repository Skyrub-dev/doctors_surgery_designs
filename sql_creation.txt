//SQL TO BE INCLUDED ON MYSQL

//CREATION OF LOGIN SYSTEM:

CREATE TABLE login (
	uid int(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
    username TEXT NOT NULL,
    password VARCHAR(128) NOT NULL,
    hashedpass VARCHAR(128) NOT NULL
);

//CREATION OF BOOKING GUEST SYSTEM:

CREATE TABLE guestbook (
    uid int(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
    name TEXT NOT NULL,
    dob DATE NOT NULL,
    email TEXT NOT NULL,
    description TEXT NOT NULL
);
